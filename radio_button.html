import javax.swing.ButtonGroup; import javax.swing.JFrame; import javax.swing.JPanel; import javax.swing.JRadioButton; class RadioButton { public static void main(String[] args) { JFrame frame = new JFrame("Radio Button Example"); frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE); JPanel panel = new JPanel(); JRadioButton radioButton1 = new JRadioButton("sharma 1"); JRadioButton radioButton2 = new JRadioButton("shatrudhan 2"); JRadioButton radioButton3 = new JRadioButton("prasad 3"); // Create a ButtonGroup to group the radio buttons together ButtonGroup buttonGroup = new ButtonGroup(); buttonGroup.add(radioButton1); buttonGroup.add(radioButton2); buttonGroup.add(radioButton3); // Add the radio buttons to the panel panel.add(radioButton1); panel.add(radioButton2); panel.add(radioButton3); frame.getContentPane().add(panel); frame.pack(); frame.setVisible(true); } }
